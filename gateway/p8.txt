import java.util.Scanner;\npublic class NQueens {\n    public static void main(String[] args) {\n        Scanner sc = new Scanner(System.in);\n        System.out.print(\"Enter the grid size (e.g., 4 for 4x4): \");\n        int n = sc.nextInt();\n        placeQueens(n);\n    }\n\n    private static void placeQueens(int gridSize) {\n        if (gridSize < 4 && gridSize != 1) {\n            System.out.println(\"No Solution available\");\n        } else {\n            int[][] board = new int[gridSize][gridSize];\n            if (placeAllQueens(board, 0)) {\n                printBoard(board);\n            } else {\n                System.out.println(\"No Solution available\");\n            }\n        }\n    }\n\n    private static boolean placeAllQueens(int[][] board, int row) {\n        if (row >= board.length) {\n            return true;\n        }\n        for (int j = 0; j < board.length; j++) {\n            if (isSafe(board, row, j)) {\n                board[row][j] = 1;\n                if (placeAllQueens(board, row + 1)) {\n                    return true;\n                }\n                board[row][j] = 0; // Backtrack\n            }\n        }\n        return false;\n    }\n\n    private static boolean isSafe(int[][] board, int row, int col) {\n        // Check Left Upper Diagonal\n        for (int i = row - 1, j = col - 1; i >= 0 && j >= 0; i--, j--) {\n            if (board[i][j] == 1) return false;\n        }\n        // Check Right Upper Diagonal\n        for (int i = row - 1, j = col + 1; i >= 0 && j < board.length; i--, j++) {\n            if (board[i][j] == 1) return false;\n        }\n        // Check same Column\n        for (int i = row - 1; i >= 0; i--) {\n            if (board[i][col] == 1) return false;\n        }\n        return true;\n    }\n\n    private static void printBoard(int[][] board) {\n        for (int row = 0; row < board.length; row++) {\n            for (int col = 0; col < board.length; col++) {\n                if (board[row][col] == 1) {\n                    System.out.print(\"Q \");\n                } else {\n                    System.out.print(\"_ \");\n                }\n            }\n            System.out.println();\n        }\n        System.out.println(\"\");\n    }\n}
